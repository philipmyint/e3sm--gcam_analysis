 15def clean_up_dataframe(df, print_to_console=False):
 16    """ 
 17    Cleans up a Pandas DataFrame from junk columns that contain only NaN values and returns this cleaned-up DataFrame.
 18    The junk columns may be a result of pd.read_fwf() separating out a single column with a long label into two columns where the second column 
 19    contains the junk values. For example, the column "rho A+B+C (g/cm^3)" may get separated to "rho A+B+C" and "(g/cm^3)", in which the latter
 20    column, which will contain only NaN entries, should be deleted.
 21
 22    Parameters:
 23        df: DataFrame to be cleaned.
 24        print_to_console: Indicates if the contents of the data should be printed to the console to monitor the progress.
 25
 26    Returns:
 27        df: DataFrame after cleaning operations have been applied.
 28    """
 29    if print_to_console:
 30        print(f'\nThe original columns of the DataFrame are:\n{df.columns}\n')
 31
 32    # Number of columns.
 33    num_columns = len(df.columns)
 34
 35    # List containing the labels of junk columns.
 36    columns_to_delete = []
 37
 38    # List containing the labels of non-junk columns.
 39    columns_to_keep = []
 40
 41    # Loop over the columns in the DataFrame.
 42    for i in np.arange(num_columns):
 43
 44        # Extract the label of the current column.
 45        label = df.columns[i]
 46
 47        # If this is a junk column, record its label and go to the next column.
 48        if df[label].isnull().values.all():
 49            columns_to_delete.append(label)
 50            if print_to_console:
 51                print(f'Column {i} contains junk; its label is {label}\n')
 52            continue
 53
 54        # If the next column is a junk column and we are not in the last column, append the label of that 
 55        # next column (with some modifications) to the label of the current one.
 56        if i < num_columns - 1 and df[df.columns[i+1]].isnull().values.all():
 57            # If the label of the junk column indicates the units of the quantity in the previous column 
 58            # placed between parentheses, extract only the parentheses and the unit label between them.
 59            start = df.columns[i+1].find("(")
 60            if start != -1:
 61                end = df.columns[i+1].find(")") + 1
 62                label += " " + df.columns[i+1][start:end]
 63            # If the label of the junk column does not contain parentheses, then append that label to that of the current column.
 64            else:
 65                label += " " + df.columns[i+1]
 66            if print_to_console:
 67                print(f'The new label of column {i} is {label}')
 68
 69        # Record the labels of all non-junk columns in the appropriate list.
 70        columns_to_keep.append(label)
 71
 72    # Delete all junk columns.
 73    for label in columns_to_delete:
 74        df.drop(label, axis=1, inplace=True)
 75    
 76    # Reassign the column labels to those that we want to keep; print these labels and return the DataFrame.
 77    df.columns = columns_to_keep
 78    if print_to_console:
 79        print(f'The new columns of the modified DataFrame are:\n{df.columns}\n')
 80    return df

 82def get_columns_without_units_in_dataframe(df):
 83    """
 84    Returns all column names in a Pandas DataFrame without the units. For example, return 'Mass' if the colum name is 'Mass (kg)'.
 85
 86    Parameters:
 87        df: The input DataFrame.
 88
 89    Returns:
 90        A list of all column names, with the units removed.
 91    """
 92    columns_without_units = []
 93    for column in df.columns:
 94        stop_index = column.find(' (')
 95        # If the column does not have unit associated with it (so that find returns a -1), just return that column unmodified.
 96        if stop_index == -1:
 97            columns_without_units.append(column)
 98        else:
 99            columns_without_units.append(column[:stop_index])
100    return columns_without_units

102def get_matching_column_in_dataframe(df, variable, all_matches=False):
103    """
104    Returns the column name(s) in a Pandas DataFrame that matches the variable.
105
106    Parameters:
107        df: The input DataFrame.
108        variable: Variable name to search for in the column names.
109        all_matches: If True, all columns that matches the variable will be returned. If False, only the first matching column will be returned.
110
111    Returns:
112        The matching column name(s), or None if no match is found.
113    """
114    all_matching_columns = []
115    for column in df.columns:
116        units_in_column = column.find(' (') != -1
117        if units_in_column and variable + ' (' in column:
118            if not all_matches:
119                return column
120            else:
121                all_matching_columns.append(column)
122        if not units_in_column and variable in column:
123            if not all_matches:
124                return column
125            else:
126                all_matching_columns.append(column)
127    if all_matching_columns:
128        return all_matching_columns
129    else:
130        return None

132def move_columns_next_to_each_other_in_dataframe(df, column1, column2):
133    """
134    Rearranges a Pandas DataFrame by moving column2 next to column1.
135
136    Parameters:
137        column1: Column next to which column2 will be moved. The order of all columns prior to column1 will be unaffected.
138        column2: Column to move next to column1.
139        df: DataFrame whose columns will be rearranged.
140
141    Returns:
142        A new DataFrame with column1 and column2 moved next to each other.
143    """
144    index = df.columns.get_loc(column1)
145    new_columns = list(df.columns[:index]) + [column1, column2] + list(df.columns[index+1:])
146    df = df[new_columns]
147    # Remove the duplicate column2, keeping the first occurrence.
148    df = df.loc[:, ~df.columns.duplicated()]
149    return df

151def perform_ttest(df, columns_set_1, columns_set_2, equal_var=False):
152    """ 
153    Performs a t-test for the means of two (presumed) independent data sets.
154
155    Parameters:
156        df: Pandas DataFrame containing the columns for both data sets.
157        columns_set_1: List of columns in the DataFrame for the first data set.
158        columns_set_2: List of columns in the DataFrame for the second data set.
159        equal_var: Boolean that indicates if the two samples are characterized by the same variance.
160        
161    Returns:
162        The p-value produced by the t-test.
163    """
164    set_1 = df[columns_set_1]
165    set_2 = df[columns_set_2]
166    ttest = stats.ttest_ind(set_1, set_2, equal_var)
167    return ttest.pvalue

238def write_dataframe_to_fwf(file_name, df, keep_index_column=False, width_index_column=None):
239    """ 
240    Writes the contents of a Pandas DataFrame to an output file in fixed-width format (fwf), omitting the index column if specified to do so.
241
242    Parameters:
243        file_name: Complete path and name of the output file.
244        df: DataFrame in which the data to be written to the output file are stored.
245        keep_index_column: Specifies whether to print the index column in the output file.
246        width_index_column: Specifies the number of characters that span the width of the index column.
247
248    Returns:
249        N/A.
250    """
251    # Write the contents to the file.
252    with open(file_name, 'w') as file:
253        file.write(df.__repr__())
254    
255    # Remove the index column by deleting the number of characters equal to the width of the column.
256    # This includes the space between it and the next column.
257    if (keep_index_column == False):
258
259        # If the width of the index column has not been specified in the call to this function, set it based on the number of lines in the file.
260        if not width_index_column:
261            num_lines = len(df)
262            if num_lines > 1:
263                width_index_column = int(np.log10(num_lines-1) + 2)
264            else:
265                width_index_column = 2
266
267        for line in fileinput.input(files=(file_name), inplace=True):
268            sys.stdout.write(line[width_index_column:])
