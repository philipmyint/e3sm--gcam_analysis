 48def process_inputs(inputs):
 49    """ 
 50    Processes a dictionary of inputs (keys are options, values are choices for those options) for creating spatial plots from data in NetCDF files.
 51
 52    Parameters:
 53        inputs: Dictionary containing the user plotting choice inputs for different options. This dictionary may be incomplete or have invalid values.
 54
 55    Returns:
 56        List of dictionaries, each of which specifies completed plotting options for a single variable. If the user did not select a plotting option 
 57        for a particular variable, the default choice for that plotting option will be selected.
 58    """
 59    # Turn the list of NetCDF files and their corresponding labels into a list of lists, if they are not already in that form.
 60    for input_type in ['netcdf_files']:
 61        if isinstance(inputs[input_type], str):
 62            inputs[input_type] = [[inputs[input_type]]]
 63        elif isinstance(inputs[input_type], list) and not check_is_list_of_lists(inputs[input_type]):
 64            inputs[input_type] = [inputs[input_type]]
 65
 66    # Read one of the NetCDF files into an xarray Dataset so that we can later get the variables contained in them and the units of these variables.
 67    ds = xr.open_dataset(inputs['netcdf_files'][0][0])
 68
 69    # If the user entered the string 'all' for the variables or no input at all for the variables, assume that they want to make plots for
 70    # all variables that are in the Dataset.
 71    if 'variables' not in inputs:
 72        inputs['variables'] = None
 73    variables = inputs['variables']
 74    if not variables or variables == 'all':
 75        variables = list(ds.keys())
 76        inputs['variables'] = variables
 77    # If the user entered a string to indicate a single variable, put that string in a list.
 78    if isinstance(variables, str):
 79        variables = [variables]
 80        inputs['variables'] = variables
 81
 82    # For the plotting options that have not been specified in the inputs dictionary, add keys for them if necessary and use the default values.
 83    for key in default_inputs_spatial_data.keys():
 84        if key not in inputs:
 85            inputs[key] = default_inputs_spatial_data[key]
 86
 87    # If the user specified anything other than a dictionary (e.g., a single value [string, integer, float] or a list) for the other plotting options, 
 88    # assume that they want to use that value/list for all the variables. Enable this by creating dictionaries with the keys given by the variables. 
 89    # This also covers the case above, where default values were added for all plotting options that are not specified in the inputs dictionary.
 90    for key, value in inputs.items():
 91        if key != 'variables':
 92            if not isinstance(value, dict):
 93                inputs[key] = dict.fromkeys(variables, value)
 94    
 95    # For each variable, if a plotting option is missing (has not been specified), fill in with the default corresponding to that plotting option.
 96    if 'plot_name' not in inputs:
 97        inputs['plot_name'] = {}
 98    if 'title' not in inputs:
 99        inputs['title'] = {}
100    for variable in variables:
101        # Use the default for the plot directory and create the directory if it does not already exist.
102        if not any(key == variable for key in inputs['plot_directory'].keys()):
103            inputs['plot_directory'][variable] = default_inputs_spatial_data['plot_directory']
104        if not os.path.exists(inputs['plot_directory'][variable]):
105            os.makedirs(inputs['plot_directory'][variable])
106        # Default for the plot names is to call it 'spatial_[var_name]', where '[var_name]' is the name of the variable.
107        if not any(key == variable for key in inputs['plot_name'].keys()):
108            inputs['plot_name'][variable] = os.path.join(inputs['plot_directory'][variable], 'spatial_' + variable)
109        # Default for the title of a variable is to use the column header for that variable from the Dataset.
110        if not any(key == variable for key in inputs['title'].keys()):
111            # Replace ^2 with $^2$ in the units for the variable, so that the intended exponent (e.g., m^2) gets rendered correctly.
112            units = ds[variable].attrs['units'].replace('^2', '$^2$')
113            # Replace /m2 with /m$^2$ in the units for the variable, so that it gets rendered correctly.
114            units = units.replace('/m2', '/m$^2$')
115            inputs['title'][variable] = rf'{variable} ({units})'
116        # Default for the other plotting options are specified in the default_inputs_spatial_data dictionary.
117        for key, value in inputs.items():
118            if key not in ['variables', 'plot_directory', 'plot_name', 'title']:
119                if not any(value_key == variable for value_key in value.keys()):
120                    inputs[key][variable] = default_inputs_spatial_data[key]
121
122    # Now that the dictionary has been populated with complete plotting options for each variable, separate it into a list of dictionaries,
123    # where each of these smaller dictionaries contain the plotting options for a single variable. Return this list of dictionaries.
124    list_of_inputs = []
125    for variable in inputs['variables']:
126        inputs_for_this_variable = {'variable': variable}
127        for input_option in inputs.keys():
128            if input_option != 'variables':
129                inputs_for_this_variable[input_option] = inputs[input_option][variable]
130        list_of_inputs.append(inputs_for_this_variable)
131    return list_of_inputs

133def plot_spatial_data_eam(inputs, grid_file):
134    """ 
135    Parses a dictionary of inputs (keys are options, values are choices for those options) to create spatial plots for a single variable from E3SM EAM
136    outputs. The data for these spatial plots are stored in NetCDF files specified by the inputs dictionary.
137
138    Parameters:
139        input: Dictionary containing the user plotting choice inputs for different options. This dictionary is assumed to be complete (pre-processed).
140        grid_file: Path and name of the grid file for the EAM unstructured mesh.
141
142    Returns:
143        N/A.
144    """
145    # This function creates spatial plots for a single variable, and so it assumes that there is only one variable in the inputs dictionary.
146    start_time = time.time()
147    variable = inputs['variable']
148
149    # Extract all other plotting options.
150    plot_directory = inputs['plot_directory']
151    multiplier = inputs['multiplier']
152    netcdf_files = inputs['netcdf_files']
153    cmap_color = inputs['cmap']
154    projection = inputs['projection']
155    title = inputs['title'] 
156    title_size = inputs['title_size']
157    width = inputs['width'] 
158    height = inputs['height']
159    plot_name = inputs['plot_name']
160    cbar_on = inputs['cbar_on']
161    cbar_label_size = inputs['cbar_label_size']
162    cbar_limits = inputs['cbar_limits']
163    cbar_x_offset = inputs['cbar_x_offset']
164    statistics_panel_size = inputs['statistics_panel_size']
165    calculation_type = inputs['calculation_type']
166    plot_type_for_2_sets = inputs['plot_type_for_2_sets']
167    use_latex = inputs['use_latex']
168    start_year = inputs['start_year']
169    end_year = inputs['end_year']
170    p_value_threshold = inputs['p_value_threshold']
171    p_value_file = inputs['p_value_file']
172    p_value_file_print_only_if_below_threshold = inputs['p_value_file_print_only_if_below_threshold']
173    stippling_on = inputs['stippling_on']
174    stippling_std_multiple = inputs['stippling_std_multiple']
175    stippling_hatches = inputs['stippling_hatches']
176
177    # Store the grid file in an uxarray Dataset.
178    grid = ux.open_grid(grid_file)
179
180    # Read each of the NetCDF output files, which are arranged in a list of lists (2D matrix), into an uxarray DataArray and then add each of these 
181    # DataArrays to a single uxarray Dataset that will store the data from all of the files. To form the DataArrays, calculate either the mean or sum 
182    # between the start and end years for each lat/lon coordinate. We will later display some function of this mean or sum on the spatial plot.
183    num_files_in_each_set = len(netcdf_files)
184    num_file_sets = len(netcdf_files[0])
185    uxds = ux.UxDataset()
186    for file_set_index in range(num_file_sets):
187        for file_index in range(num_files_in_each_set):
188            file = netcdf_files[file_index][file_set_index]
189            # Create a temporary NetCDF file with data between only the start and end years. 
190            ds = xr.open_dataset(file).sel(year=slice(start_year, end_year))[variable]
191            file = os.path.join(plot_directory, f'temp_{variable}.nc')
192            ds.to_netcdf(file, 'w')
193            if calculation_type == 'mean':
194                uxda = ux.open_dataset(grid_file, file).mean(dim='year')[variable]*multiplier
195            elif calculation_type == 'sum':
196                uxda = ux.open_dataset(grid_file, file).sum(dim='year')[variable]*multiplier
197                # If calculating the sum, change the per-time quantities and their units accordingly.
198                per_time_labels = ['/year', '/month', '/day', '/hour', '/min', '/s']
199                time_multipliers = np.array([1, years_TO_months, years_TO_days, years_TO_hours, years_TO_mins, years_TO_s])
200                for index, per_time_label in enumerate(per_time_labels):
201                    if per_time_label in title:
202                        title = title.replace(per_time_label, '')
203                        uxda.attrs['units'] = uxda.attrs['units'].replace(per_time_label, '')
204                        uxda *= time_multipliers[index]
205                        break
206            # If there is more than one data set, modify the labels so that we know which data set corresponds to which variable of the DataFrame.
207            if num_file_sets == 2:
208                uxda = uxda.rename(f'{variable}_{file_index}_{file_set_index}')
209            uxds[uxda.name] = uxda
210            # Delete the temporary NetCDF file now that the data have been read.
211            os.system(f'rm {file}')
212    
213    # Initialize list that will store all the uxarray DataArrays that we will want to plot for the variable.
214    uxDataArrays_to_plot = []
215    df = uxds.to_dataframe()
216    if num_file_sets == 2:
217        columns_control_set = [column for column in df.columns if column.endswith(f'_0')]
218        columns_test_set = [column for column in df.columns if column.endswith(f'_1')]
219        # Perform a t-test to compare the two spatial data sets as whole over all coordinates. Print the results to the console and to an output file.
220        df_control_set = df[columns_control_set].mean(axis=1)
221        df_test_set = df[columns_test_set].mean(axis=1)
222        ttest = stats.ttest_ind(df_control_set, df_test_set)
223        print_p_values(ttest, variable, p_value_threshold, p_value_file, plot_directory, p_value_file_print_only_if_below_threshold)
224        if plot_type_for_2_sets == 'absolute_difference':
225            # Plot absolute differences between the two data sets. 
226            df = df_test_set - df_control_set
227            uxDataArrays_to_plot.append(convert_xarray_to_uxarray(df.to_xarray(), grid, variable=variable))
228        elif plot_type_for_2_sets == 'percent_difference':
229            # Plot percent differences between the two data sets. Add a tiny number to avoid a divide-by-zero error. Take the absolute value
230            # so that if the control is negative, while the test set is positive, we get a positive value for the percent difference.
231            df = ((df_test_set - df_control_set)/(df_control_set.abs() + EPSILON))*100
232            uxDataArrays_to_plot.append(convert_xarray_to_uxarray(df.to_xarray(), grid, variable=variable))
233            title = replace_inside_parentheses(title, rf'($\%$ difference)')
234        elif plot_type_for_2_sets == 'separate_plots':
235            # Plot the two data sets individually in their own separate plots. 
236            uxDataArrays_to_plot.append(convert_xarray_to_uxarray(df_control_set.to_xarray(), grid, variable=variable))
237            uxDataArrays_to_plot.append(convert_xarray_to_uxarray(df_test_set.to_xarray(), grid, variable=variable))
238    elif num_file_sets == 1:
239        # If we have a single data set (but potentially multiple files in this data set), take the mean over all files for each lat/lon coordinate.
240        df = df.mean(axis=1)
241        uxDataArrays_to_plot.append(convert_xarray_to_uxarray(df.to_xarray(), grid, variable=variable))
242
243    # If stippling_on is True, meaning we want to add markers on plot to indicate potential regions of statistical significance, 
244    if stippling_on:
245        uxds['lat'] = grid.face_lat
246        uxds['lon'] = grid.face_lon
247        # If there is more than one file per data set and we do not want separate plots, perform a t-test at each individual lat/lon coordinate.
248        if num_files_in_each_set >= 2 and plot_type_for_2_sets != 'separate_plots':
249            df = uxds.to_dataframe()
250            df = df.groupby(['lat', 'lon']).mean()
251            da_pvalues = df.apply(perform_ttest, columns_set_1=columns_control_set, columns_set_2=columns_test_set, axis=1).fillna(1).to_xarray().fillna(1)
252            #uxda_pvalues = convert_xarray_to_uxarray(da_pvalues, grid, variable=variable, fillna=1)
253
254    # Iterate over all uxDataArrays in the list to create a plot for each one.
255    for uxda_index, uxda in enumerate(uxDataArrays_to_plot):
256
257        # Calculate some basic statistics of the current uxDataArray.
258        min, mean, median, max, std = calculate_statistics_of_xarray(uxda, variable)
259
260        # Use LaTeX for the labels if specified to do so.
261        if use_latex:
262            plt.rc('text', usetex=True)
263            plt.rc('font', family='serif', weight='bold') 
264        
265        # If plotting a percent difference and no colorbar limits are specified, set them to be -100 and 100 percent if the max exceeds 100 percent.
266        if plot_type_for_2_sets == 'percent_difference' and not cbar_limits and max > 100:
267            cbar_limits = [-100, 100]
268
269        # Plot the uxDataArray and optionally add the title and colorbar.
270        fig = plt.figure(figsize=(width, height))
271        ax = fig.add_axes([0.1, 0.1, 0.8, 0.8], projection=projection())
272        uxda = uxda[variable]
273        uxda_fig = uxda.to_polycollection(cache=True)
274        uxda_fig.set_transform(ccrs.PlateCarree())
275        uxda_fig.set_cmap(cmap_color)
276        ax.set_title(title)
277        ax.add_collection(uxda_fig)
278        ax.set_global()
279        if title:
280            plt.rcParams['axes.titlesize'] = title_size
281            ax.set_title(title)
282        if cbar_on:
283            # Extra for the colorbar padding in case using the default (not LaTeX) font.
284            if not use_latex and cbar_x_offset < 0.09:
285                cbar_x_offset = 0.09
286            cbar_ax = fig.add_axes([ax.get_position().x1+cbar_x_offset, ax.get_position().y0, 0.02, ax.get_position().height])
287            if cbar_limits:
288                uxda_fig.set_clim(vmin=cbar_limits[0], vmax=cbar_limits[1])
289            cbar = plt.colorbar(uxda_fig, cax=cbar_ax)
290            cbar.ax.tick_params(labelsize=cbar_label_size, length=0)
291
292        # Add stippling to indicate regions of potential statistical significance. 
293        if stippling_on:
294            plt.rcParams['hatch.linewidth'] = 0.5
295            plt.rcParams['hatch.color'] = 'gray'
296            if num_file_sets == 2 and num_files_in_each_set >= 2 and plot_type_for_2_sets != 'separate_plots':
297                # If there are two data sets and at least two files in each data set so that we will have previously calculated p-values at each
298                # lat/lon coordinate, the stippling will indicate regions where the p-value is less than the designated threshold.
299                mask = da_pvalues <= p_value_threshold
300                #print(mask)
301                ax.contourf(da_pvalues.lon, da_pvalues.lat, mask, levels=1, hatches=['', stippling_hatches], 
302                            alpha=0, transform=ccrs.PlateCarree())
303            else:
304                # For all other cases, the stippling will indicate regions where the value is +/- some multiple of the standard deviation 
305                # (default is 2*std) away from the mean.
306                mask = np.abs(uxda) >= mean + stippling_std_multiple*std
307                #print(uxds)
308                ax.contourf(uxds['lon'], uxds['lat'], mask, levels=1, hatches=['', stippling_hatches], alpha=0, transform=ccrs.PlateCarree())
309        
310        # Display statistics.
311        ax.text(x=0.88, y=0.9, s=f'Max:{max:.2e}\nMean:{mean:.2e}\nMedian:{median:.2e}\nMin:{min:.2e}', ha='left', 
312                fontsize=statistics_panel_size, transform=ax.transAxes)
313        ax.text(x=0.88, y=0.05, s=f'Std:{std:.2e}', ha='left', fontsize=statistics_panel_size, transform=ax.transAxes)
314
315        # Add additional features like coastline and oceans.
316        ax.coastlines(lw=0.6)
317
318        # Save the figure and then close it. Record the elapsed time.
319        end_time = time.time()
320        elapsed_time = end_time - start_time
321        if len(uxDataArrays_to_plot) > 1:
322            # If we want to plot two data sets separately, add a set number to the figure name for each set.
323            save_figure(plot_name + f'_set_{uxda_index+1}', fig, inputs)
324            print(f"Elapsed time for producing plots for {variable} (set {uxda_index+1}) in {plot_directory}: {elapsed_time:.2f} seconds") 
325        else:
326            save_figure(plot_name, fig, inputs)
327            print(f"Elapsed time for producing plots for {variable} in {plot_directory}: {elapsed_time:.2f} seconds") 
328        plt.close(fig)
